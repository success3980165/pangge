使用 ~~ 和 0| 来代替 Math.Floor 处理正数

$(function() {
    var
    num_1  = Math.random()*10,
    num_2  = Math.floor(3.1415926),
    rand_1 = Math.floor(Math.random()*10), // before
    rand_2 = 0|Math.random()*10, // after
    rand_3 = ~~(Math.random()*10); //或者这样

    console.log(num_1,num_2);
    console.log(rand_1,rand_2,rand_3);

    //~~是一种利用符号进行的类型转换,转换成数字类型
    ~~true == 1
    ~~false == 0
    ~~"" == 0
    ~~[] == 0

    ~~undefined ==0
    ~~!undefined == 1
    ~~null == 0
    ~~!null == 1
});
