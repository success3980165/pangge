可以给 setTimeout 和 setInterval 传递参数

$(function() {
    //假设你想给 setTimeout 中调用的函数传递两个参数，你可能会觉得这是唯一的方法：
    var callback = function(a, b ,c) {
        console.log(a + b + c); // 'foobar'
    }

    window.setTimeout(function() {
        callback('xia', 'yu', 'ting');
    }, 1000);


    //实际上，你还可以用下面的方法实现同样的功能：
    //注：该方法不支持IE（包括IE9）。
    var
    foo = 10,
    bar = 20;
    window.setTimeout(callback, 1000, foo, bar, 0.1);
});
